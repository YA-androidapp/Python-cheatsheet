```py
from datetime import datetime

from collections import OrderedDict
from faker import Faker
import csv

with open('test-fileio/output_faker.csv', 'w+', encoding='utf-8', newline='') as f:
    csv_writer = csv.writer(f)
    fake = Faker()
    # fake_jp = Faker('ja_JP')
    for i in range(100):
        l = [

            fake.ean(),
            fake.ean(length=8),

            fake.ean13(),
            fake.ean13(leading_zero=False),
            fake.ean13(leading_zero=True),

            fake.ean8(),

            fake.upc_a(),
            fake.upc_a(upc_ae_mode=True, base='120004', number_system_digit=0),

            fake.upc_e(),
            fake.upc_e(base='120004', number_system_digit=0, safe_mode=False)

        ]
        csv_writer.writerow(l)
        if 0 == i % 100:
            t = [datetime.now().strftime("%Y/%m/%d %H:%M:%S.%f")]
            csv_writer.writerow(t)
```
